# Extra dependencies for building voxel phantoms from real segmented CT data
# (services/compute/app/build_chest.py). Heavy — install into the same venv:
#
#   ./.venv/Scripts/python.exe -m pip install torch --index-url https://download.pytorch.org/whl/cpu
#   ./.venv/Scripts/python.exe -m pip install -r requirements-models.txt
#
# Then build the bundled human-chest phantom from the 3D Slicer CTChest sample:
#   # 1. download the sample CT
#   curl -L -o data/chest/CT-chest.nrrd \
#     https://github.com/Slicer/SlicerTestingData/releases/download/SHA256/4507b664690840abb6cb9af2d919377ffc4ef75b167cb6fd0f747befdb12e38e
#   # 2. convert + segment every organ (TotalSegmentator; --fast ~3mm, drop it for full res)
#   ./.venv/Scripts/python.exe -c "import SimpleITK as s; s.WriteImage(s.ReadImage('data/chest/CT-chest.nrrd'),'data/chest/ct.nii.gz')"
#   ./.venv/Scripts/TotalSegmentator.exe -i data/chest/ct.nii.gz -o data/chest/seg.nii --ml --fast
#   # 3. build the material-map voxel volume + display mesh
#   ./.venv/Scripts/python.exe -m app.build_chest --ct data/chest/ct.nii.gz --seg data/chest/seg.nii \
#     --out ../../apps/web/public/models/chest --spacing 1.0
#
# ---------------------------------------------------------------------------
# Additional anatomical models — head/neck, chest-abdo-pelvis, upper/lower
# extremity, whole body — are built from the TotalSegmentator dataset (small
# 102-case subset, ships ground-truth segmentations so no inference is needed):
#   https://zenodo.org/records/10047263  (Totalsegmentator_dataset_small_v201.zip, 3.2 GB)
# One "ct whole body" case (e.g. s1397) yields every region via cranio-caudal /
# 3D-bbox crops. Combine its per-structure masks then build each region:
#   D=/path/to/Totalsegmentator_dataset_small_v201
#   ./.venv/Scripts/python.exe -m app.combine_seg --case $D/s1397 --out data/wb/s1397_seg.nii.gz
#   for r in headneck chestabdopelvis upperextremity lowerextremity wholebody; do
#     ./.venv/Scripts/python.exe -m app.build_model --ct $D/s1397/ct.nii.gz \
#       --seg data/wb/s1397_seg.nii.gz --out ../../apps/web/public/models/$r \
#       --name $r --region $r --spacing 1.5 --source "TotalSegmentator dataset (s1397)"
#   done
# NB clinical whole-body CT spans vertex→pelvis, so the extremities are proximal
# (shoulder/upper-arm, hip/femur); distal limbs need dedicated extremity CT.
#
# ---------------------------------------------------------------------------
# TRUE head-to-toe whole body + distal extremities: the TotalSegmentator subset
# above stops at the pelvis. Use the VSD Full Body collection instead — open
# anonymized whole-body POSTMORTEM forensic CT (Bern/Zürich), already NIfTI,
# CC BY-NC-SA (non-commercial; fine for this educational sim):
#   https://zenodo.org/records/8270365   (DOI 10.5281/zenodo.8270364; 109 subjects)
#   Direct per-subject: https://zenodo.org/records/8270365/files/<id>.zip?download=1
# The z-prefixed ids (z033, z045, …) are single whole-body volumes; plain-numbered
# ids are separate regional DICOM series. CAUTION: forensic positioning VARIES —
# many subjects are flexed (folded hips/knees). Inspect first and pick an EXTENDED
# supine subject (z045 = 45M, extended, arms abducted, 512×512×3117 @1.0/0.5mm,
# 1558 mm head-to-feet is a good one):
#   ./.venv/Scripts/python.exe -m app.inspect_vsd data/vsd/z045.zip data/vsd/z045 z045.png
#   # squeeze the 4D NIfTI -> ct.nii.gz (see prep step inside), then segment IF the
#   # torch/torchvision versions match (a mismatch -> "torchvision::nms does not
#   # exist"; TS's NN then can't run). materialize() derives air/lung/fat/muscle/
#   # bone/skin from HU ALONE, so a ZERO-label seg still yields a full attenuation
#   # model (only organ mesh colours are lost):
#   ./.venv/Scripts/python.exe -c "import nibabel as nib,numpy as np; c=nib.load('data/vsd/z045/ct.nii.gz'); nib.save(nib.Nifti1Image(np.zeros(c.shape[:3],np.uint8),c.affine),'data/vsd/z045/seg.nii')"
#   ./.venv/Scripts/python.exe -m app.build_model --ct data/vsd/z045/ct.nii.gz \
#     --seg data/vsd/z045/seg.nii --out ../../apps/web/public/models/wholebody \
#     --name wholebody --region wholebody --spacing 2.0 --source "VSD Full Body (z045)"
# TODO: extremity crops from the extended subject (LE: femur→feet z-window + single
# leg; UE: abducted-arm bbox to the hand) + restore organ seg once torch/torchvision
# is realigned.
